// Recipes for creating objects from mining

function|script|MiningRecipes|{

    cleararray @RecipesCho$,"",getarraysize(@RecipesCho$);
    cleararray @RecipesIdx$,"",getarraysize(@RecipesIdx$);
    set @RecipeNr, 0;
    set @padcnt, 0;

    // All Recipes for mining.
    // Level 9
    if((@HasSkill > 8)) 
        goto L_NineSkills;
    // Level 8
    if((@HasSkill > 7)) 
        goto L_EightSkills;
    // Level 7
    if((@HasSkill > 6)) 
        goto L_SevenSkills;
    // Level 6
    if((@HasSkill > 5)) 
        goto L_SixSkills;
    // Level 5
    if((@HasSkill > 4)) 
        goto L_FiveSkills;
    // Level 4
    if((@HasSkill > 3)) 
        goto L_FourSkills;
    // Level 3
    if((@HasSkill > 2)) 
        goto L_ThreeSkills;
    // Level 2
    if((@HasSkill > 1)) 
        goto L_TwoSkills;
    // Level 1
    if((@HasSkill > 0)) 
        goto L_OneSkills;
    goto L_End;

    L_NineSkills:
        setarray @RecipesCho$[@RecipeNr], "Amethyst";
        setarray @RecipesIdx$[@RecipeNr], "MRAmethyst";
        set @RecipeNr, @RecipeNr + 1;
        goto L_EightSkills;
    L_EightSkills:
        setarray @RecipesCho$[@RecipeNr], "Sapphire";
        setarray @RecipesIdx$[@RecipeNr], "MRSapphire";
        set @RecipeNr, @RecipeNr + 1;
        goto L_SevenSkills;
    L_SevenSkills:
        goto L_SixSkills;
    L_SixSkills:
        goto L_FiveSkills;
    L_FiveSkills:
        goto L_FourSkills;
    L_FourSkills:
        setarray @RecipesCho$[@RecipeNr], "Terranite Ingot";
        setarray @RecipesIdx$[@RecipeNr], "MRTerraniteIngot";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Coal";
        setarray @RecipesIdx$[@RecipeNr], "MRCoal";
        set @RecipeNr, @RecipeNr + 1;
        goto L_ThreeSkills;
    L_ThreeSkills:
        setarray @RecipesCho$[@RecipeNr], "Terranite Ore";
        setarray @RecipesIdx$[@RecipeNr], "MRTerraniteOre";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Diamond";
        setarray @RecipesIdx$[@RecipeNr], "MRDiamond";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Emerald";
        setarray @RecipesIdx$[@RecipeNr], "MREmerald";
        set @RecipeNr, @RecipeNr + 1;
        goto L_TwoSkills;
    L_TwoSkills:
        setarray @RecipesCho$[@RecipeNr], "Iron Ore";
        setarray @RecipesIdx$[@RecipeNr], "MRIronOre";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Iron Ingot";
        setarray @RecipesIdx$[@RecipeNr], "MRIronIngot";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Topaz";
        setarray @RecipesIdx$[@RecipeNr], "MRTopaz";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Ruby";
        setarray @RecipesIdx$[@RecipeNr], "MRRuby";
        set @RecipeNr, @RecipeNr + 1;
        goto L_OneSkills;
    L_OneSkills:
        setarray @RecipesCho$[@RecipeNr], "Tin";
        setarray @RecipesIdx$[@RecipeNr], "MRTin";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Copper Ore";
        setarray @RecipesIdx$[@RecipeNr], "MRCopperOre";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Bronze Ingot";
        setarray @RecipesIdx$[@RecipeNr], "MRBronzeIngot";
        set @RecipeNr, @RecipeNr + 1;
        goto L_Pad;

    L_AddPadItem:
        setarray @RecipesCho$[@RecipeNr], "None";
        setarray @RecipesIdx$[@RecipeNr], "None";
        set @RecipeNr, @RecipeNr + 1;
        set @padcnt, @padcnt + 1;
        if(@paditem > @padcnt) 
            goto L_AddPadItem;
        goto L_End; 

    L_Pad:
        set @paditem, ((getarraysize(@RecipesCho$) - ((getarraysize(@RecipesCho$)/5)*5))+1);
        message strcharinfo(0), @paditem;
        goto L_AddPadItem;

    L_End:
        return;
}
