
function|script|ForgingRecipes|{

    cleararray @RecipesCho$,"",getarraysize(@RecipesCho$);
    cleararray @RecipesIdx$,"",getarraysize(@RecipesIdx$);
    set @RecipeNr, 0;
    set @padcnt, 0;

    // All Recipes for mining.
    // Level 9
    if((@HasSkill > 8)) 
        goto L_NineSkills;
    // Level 8
    if((@HasSkill > 7)) 
        goto L_EightSkills;
    // Level 7
    if((@HasSkill > 6)) 
        goto L_SevenSkills;
    // Level 6
    if((@HasSkill > 5)) 
        goto L_SixSkills;
    // Level 5
    if((@HasSkill > 4)) 
        goto L_FiveSkills;
    // Level 4
    if((@HasSkill > 3)) 
        goto L_FourSkills;
    // Level 3
    if((@HasSkill > 2)) 
        goto L_ThreeSkills;
    // Level 2
    if((@HasSkill > 1)) 
        goto L_TwoSkills;
    // Level 1
    if((@HasSkill > 0)) 
        goto L_OneSkills;
    goto L_End;

    L_NineSkills:
        setarray @RecipesCho$[@RecipeNr], "Terranite Armor"; 
        setarray @RecipesIdx$[@RecipeNr], "TerraniteChestArmor";  
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Terranite Legs"; 
        setarray @RecipesIdx$[@RecipeNr], "TerraniteLegs";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_EightSkills;
    L_EightSkills:
        setarray @RecipesCho$[@RecipeNr], "Warlord Plate"; 
        setarray @RecipesIdx$[@RecipeNr], "WarlordPlate";
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Steel Shield"; 
        setarray @RecipesIdx$[@RecipeNr], "SteelShield";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_SevenSkills;
    L_SevenSkills:
        setarray @RecipesCho$[@RecipeNr], "Terranite Helmet"; 
        setarray @RecipesIdx$[@RecipeNr], "TerraniteHelmet";  
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Warlord Boots"; 
        setarray @RecipesIdx$[@RecipeNr], "WarlordBoots";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_SixSkills;
    L_SixSkills:
        setarray @RecipesCho$[@RecipeNr], "Terranite Helmet"; 
        setarray @RecipesIdx$[@RecipeNr], "TerraniteHelmet";  
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Warlord Helmet"; 
        setarray @RecipesIdx$[@RecipeNr], "WarlordHelmet";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_FiveSkills;
    L_FiveSkills:
        setarray @RecipesCho$[@RecipeNr], "Crusade Helmet"; 
        setarray @RecipesIdx$[@RecipeNr], "CrusadeHelmet";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_FourSkills;
    L_FourSkills:
        setarray @RecipesCho$[@RecipeNr], "Light Platemail"; 
        setarray @RecipesIdx$[@RecipeNr], "LightPlatemail";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_ThreeSkills;
    L_ThreeSkills:
        setarray @RecipesCho$[@RecipeNr], "Knights Helmet"; 
        setarray @RecipesIdx$[@RecipeNr], "KnightsHelmet";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_TwoSkills;
    L_TwoSkills:
        setarray @RecipesCho$[@RecipeNr], "Chain Mail"; 
        setarray @RecipesIdx$[@RecipeNr], "ChainMail";  
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Short Sword"; 
        setarray @RecipesIdx$[@RecipeNr], "ShortSword";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_OneSkills;
    L_OneSkills:
        setarray @RecipesCho$[@RecipeNr], "Sharp Knife"; 
        setarray @RecipesIdx$[@RecipeNr], "SharpKnife";  
        set @RecipeNr, @RecipeNr + 1;
        setarray @RecipesCho$[@RecipeNr], "Knife";  
        setarray @RecipesIdx$[@RecipeNr], "Knife";  
        set @RecipeNr, @RecipeNr + 1;
        goto L_Pad;

    L_AddPadItem:
        setarray @RecipesCho$[@RecipeNr], "None";
        setarray @RecipesIdx$[@RecipeNr], "None";
        set @RecipeNr, @RecipeNr + 1;
        set @padcnt, @padcnt + 1;
        if(@paditem > @padcnt) 
            goto L_AddPadItem;
        goto L_End; 

    L_Pad:
        set @paditem, ((getarraysize(@RecipesCho$) - ((getarraysize(@RecipesCho$)/5)*5))+1);
        goto L_AddPadItem;

    L_End:
        return;
}
