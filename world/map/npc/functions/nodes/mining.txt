// Template to call mining veins
// written by wushin

function|script|CoalVein|{
    set @NodeDrop$, "Coal";
    set @NodeLevel, 1;
    set @NodeHitCount, 3;
    set @NodeXp, 1;
    set @RequiredSkill, SKILL_MINING;
    set @RequiredSkillName$, "Mining";
    set @RequiredItem$, "MiningPick";
    set @NodeReduceAttr, bStr;
    set @DROP_CHNACE_EPIC, 1000;
    set @FOUND_WAIT_TIME_BASE, 15;
    setarray @BonGatherRewards$, "Pile Of Ash","Sulphur Powder";
    callfunc("GatherEvent");
}
