function|script|CloverPatch|{
    set @NodeDrop$, "Grass Seeds";
    set @NodeLevel, 1;
    set @NodeHitCount, 3;
    set @NodeXp, 1;
    set @RequiredSkill, SKILL_FORGING;
    set @RequiredSkillName$, "Forging";
    set @RequiredItem$, "PruningKnife";
    set @NodeReduceAttr, bInt;
    set @DROP_CHNACE_EPIC, 10000;
    set @FOUND_WAIT_TIME_BASE, 15;
    setarray @BonGatherRewards$, "Four-Leaf Clover";
    callfunc("GatherEvent");
}
