// Spawns mobs

019-3.gat,0,0,0|script|SpawnManager019_3|-1,
{
    end; // just to be secure
OnCommandXmasSpawn:
OnInit:
    set $@019_3_CANDIEDSLIME_MONSTER_MAX_NUMBER,10;
    set $@019_3_CANDIEDSLIME_MOMSTER_ID,1111;
    set $@019_3_CANDIEDSLIME_MONSTER_SPAWN_X1,24;
    set $@019_3_CANDIEDSLIME_MONSTER_SPAWN_Y1,24;
    set $@019_3_CANDIEDSLIME_MONSTER_SPAWN_X2,123;
    set $@019_3_CANDIEDSLIME_MONSTER_SPAWN_Y2,101;
    if($@xmas_time)
        goto L_Init_Spawn;
    goto L_StopTracking;

L_Init_Spawn:
    initnpctimer;
    areamonster "019-3.gat", $@019_3_CANDIEDSLIME_MONSTER_SPAWN_X1, $@019_3_CANDIEDSLIME_MONSTER_SPAWN_Y1, $@019_3_CANDIEDSLIME_MONSTER_SPAWN_X2, $@019_3_CANDIEDSLIME_MONSTER_SPAWN_Y2, "Santa Slime", $@019_3_CANDIEDSLIME_MOMSTER_ID, $@019_3_CANDIEDSLIME_MONSTER_MAX_NUMBER, "SpawnManager::OnMonsterDeath";
    set $@monster_number,$@019_3_CANDIEDSLIME_MONSTER_MAX_NUMBER;
    end;

L_Spawn:
    initnpctimer;
    areamonster "019-3.gat", $@019_3_CANDIEDSLIME_MONSTER_SPAWN_X1, $@019_3_CANDIEDSLIME_MONSTER_SPAWN_Y1, $@019_3_CANDIEDSLIME_MONSTER_SPAWN_X2, $@019_3_CANDIEDSLIME_MONSTER_SPAWN_Y2, "Santa Slime", $@019_3_CANDIEDSLIME_MOMSTER_ID, 1, "SpawnManager::OnMonsterDeath";
    set $@019_3_CANDIEDSLIME_monster_number,$@019_3_CANDIEDSLIME_monster_number+1;
    end;

OnTimer20000:
    if($@xmas_time)
        goto L_Spawn;
    goto L_StopTracking;

L_StopTracking:
    stopnpctimer;
    disablenpc "SpawnManager019_3";
    end;

OnMonsterDeath:
    set $@019_3_CANDIEDSLIME_monster_number,$@019_3_CANDIEDSLIME_monster_number-1;
    end;
}
