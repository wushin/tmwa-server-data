function|script|ReplaceFlower|,
{
    if($@VDAY_FAIRY_STATUS)
        goto L_Fairy;
    goto L_Plant;

L_Fairy:
    disablenpc "#Vday Flower";
    enablenpc "#Vday Fairy";
    goto L_Return;

L_Plant:
    disablenpc "#Vday Fairy";
    enablenpc "#Vday Flower";
    goto L_Return;

L_Return:
    return;
}
